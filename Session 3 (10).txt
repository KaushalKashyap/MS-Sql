SELECT DEPT, SUM(SALARY) AS 'TOTALSAL'
FROM EMP_SAL
GROUP BY DEPT;

SELECT DEPT, DESI,COUNT(EID) AS 'TEAM_MEMBERS', SUM(SALARY) AS 'TOTALSAL'
FROM EMP_SAL
GROUP BY DEPT, DESI;

SELECT DEPT, SUM(SALARY) AS 'TOTALSAL'
FROM EMP_SAL
GROUP BY DEPT
HAVING SUM(SALARY) > 1500000;

SELECT COUNT(EID)AS 'NO OF MEMBERS', SUM (SALARY) AS 'TOTAL SAL' 
FROM EMP_SAL;

SELECT DEPT,COUNT(EID)AS 'NO OF MANAGERS', SUM (SALARY) AS 'TOTAL SAL' 
FROM EMP_SAL
WHERE DESI = 'MANAGER'
GROUP BY DEPT
HAVING SUM(SALARY) > 200000
ORDER BY SUM(SALARY) DESC;


SELECT DEPT,COUNT(EID)AS 'NO OF MANAGERS', SUM (SALARY) AS 'TOTAL SAL' 
FROM EMP_SAL
WHERE DESI = 'MANAGER' AND SALARY > 200000
GROUP BY DEPT;
